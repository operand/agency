  8class Space(ABC, metaclass=ABCMeta):
  9    """
 10    A Space is responsible for:
 11    - managing the connection lifecycle of its agents
 12    - routing messages between agents
 13    """

 15    def __init__(self, processor_class: type = NativeThreadProcessor):
 16        self.__processor_class = processor_class
 17        self.__agent_processors: Dict[str, dict] = {}
 18
 19    def add(self, agent: Agent):
 20        """
 21        Adds an agent to the space allowing it to communicate
 22        """
 23        try:
 24            self._connect(agent)
 25
 26            def process():
 27                self._consume(agent)
 28
 29            processor = self.__processor_class(process)
 30            self.__agent_processors[agent.id()] = {
 31                "agent": agent,
 32                "processor": processor,
 33            }
 34            agent._space = self
 35            agent.after_add()
 36            processor.start()
 37        except:
 38            # clean up and raise if an error occurs
 39            self.remove(agent)
 40            raise
 41
 42    def remove(self, agent: Agent):
 43        """
 44        Removes an agent from the space.
 45        """
 46        agent.before_remove()
 47        ap = self.__agent_processors.pop(agent.id(), None)
 48        if ap is not None:
 49            ap['processor'].stop()
 50        self._disconnect(agent)
 51        agent._space = None
 52
 53    def remove_all(self):
 54        """
 55        Removes all agents added through this space instance.
 56        """
 57        agents = [ap['agent'] for ap in self.__agent_processors.values()]
 58        for agent in agents:
 59            self.remove(agent)
 60        self.__agent_processors.clear()

 63        """
 64        Validates and delivers a message to the appropriate agents
 65        """
 66        message = Message(**message).dict(

 68            exclude_unset=True,
 69        )
 70        self._deliver(message)

102        """
