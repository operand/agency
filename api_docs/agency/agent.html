 16def __generate_help(method: callable) -> dict:
 17    """
 18    Generates a help object from a method's docstring and signature
 19
 20    Args:
 21        method: the method
 22
 23    Returns:
 24        A help object of the form:
 25
 26        {
 27            "description": <description>,
 28            "args": {
 29                "arg_name": {
 30                    "type": <type>,
 31                    "description": <description>
 32                },
 33            }
 34            "returns": {
 35                "type": <type>,
 36                "description": <description>
 37            }
 38        }
 39    """
 40    signature = inspect.signature(method)
 41    parsed_docstring = parse(method.__doc__, DocstringStyle.GOOGLE)
 42
 43    help_object = {}
 44
 45    # description
 46    if parsed_docstring.short_description is not None:
 47        description = parsed_docstring.short_description
 48        if parsed_docstring.long_description is not None:
 49            description += " " + parsed_docstring.long_description
 50        help_object["description"] = re.sub(r"\s+", " ", description).strip()
 51
 52    # args
 53    help_object["args"] = {}
 54    docstring_args = {arg.arg_name: arg for arg in parsed_docstring.params}
 55    arg_names = list(signature.parameters.keys())[1:]  # skip 'self' argument
 56    for arg_name in arg_names:
 57        arg_object = {}
 58
 59        # type
 60        sig_annotation = signature.parameters[arg_name].annotation
 61        if sig_annotation is not None and sig_annotation.__name__ != "_empty":
 62            arg_object["type"] = util.python_to_json_type_name(
 63                signature.parameters[arg_name].annotation.__name__)
 64        elif arg_name in docstring_args and docstring_args[arg_name].type_name is not None:
 65            arg_object["type"] = util.python_to_json_type_name(
 66                docstring_args[arg_name].type_name)
 67
 68        # description
 69        if arg_name in docstring_args and docstring_args[arg_name].description is not None:
 70            arg_object["description"] = docstring_args[arg_name].description.strip()
 71
 72        help_object["args"][arg_name] = arg_object
 73
 74    # returns
 75    if parsed_docstring.returns is not None:
 76        help_object["returns"] = {}
 77
 78        # type
 79        if signature.return_annotation is not None:
 80            help_object["returns"]["type"] = util.python_to_json_type_name(
 81                signature.return_annotation.__name__)
 82        elif parsed_docstring.returns.type_name is not None:
 83            help_object["returns"]["type"] = util.python_to_json_type_name(
 84                parsed_docstring.returns.type_name)
 85
 86        # description
 87        if parsed_docstring.returns.description is not None:
 88            help_object["returns"]["description"] = parsed_docstring.returns.description.strip()
 89
 90    return help_object

 93def action(*args, **kwargs):
 94    def decorator(method):
 95        method.action_properties = {
 96            "name": method.__name__,
 97            "access_policy": ACCESS_PERMITTED,
 98            "help": __generate_help(method),
 99            **kwargs,
100        }
101        return method
102
103    if len(args) == 1 and callable(args[0]) and not kwargs:
104        return decorator(args[0])  # The decorator was used without parentheses
105    else:
106        return decorator  # The decorator was used with parentheses

114    def __init__(self, id: str, receive_own_broadcasts: bool = True) -> None:
115        if len(id) < 1 or len(id) > 255:
116            raise ValueError("id must be between 1 and 255 characters")
117        if re.match(r"^amq\.", id):
118            raise ValueError("id cannot start with \"amq.\"")
119        if id == "*":
120            raise ValueError("id cannot be \"*\"")
121        self.__id: str = id
122        self.__receive_own_broadcasts = receive_own_broadcasts
123        self._space = None  # set by Space when added
124        self._message_log: List[Message] = []  # stores all messages
125
126    def id(self) -> str:
127        """
128        Returns the id of this agent
129        """
130        return self.__id
131
132    def send(self, message: dict):
133        """
134        Sends (out) a message
135        """
136        message["from"] = self.id()
137        self._message_log.append(message)
138        self._space._route(message)
139
140    def _receive(self, message: dict):
141        """
142        Receives and processes an incoming message
143        """
144        if not self.__receive_own_broadcasts \
145           and message['from'] == self.id() \
146           and message['to'] == '*':
147            return
148
149        try:
150            # Record message and commit action
151            self._message_log.append(message)
152            self.__commit(message)
153        except Exception as e:
154            # Here we handle exceptions that occur while committing an action,
155            # including PermissionError's from access denial, by reporting the
156            # error back to the sender.
157            self.send({
158                "to": message['from'],
159                "from": self.id(),
160                "action": {
161                    "name": "error",
162                    "args": {
163                        "error": f"{e}",
164                        "original_message_id": message.get('id'),
165                    }
166                }
167            })
168
169    def __commit(self, message: dict):
170        """
171        Invokes action if permitted otherwise raises PermissionError
172        """
173        # Check if the action method exists
174        try:
175            action_method = self.__action_method(message["action"]["name"])
176        except KeyError:
177            # the action was not found
178            if message['to'] == self.id():
179                # if it was point to point, raise an error
180                raise AttributeError(
181                    f"\"{message['action']['name']}\" not found on \"{self.id()}\"")
182            else:
183                # broadcasts will not raise an error
184                return
185
186        self.before_action(message)
187
188        return_value = None
189        error = None
190        try:
191
192            # Check if the action is permitted
193            if self.__permitted(message):
194
195                # Invoke the action method
196                # (set _current_message so that it can be used by the action)
197                self._current_message = message
198                return_value = action_method(**message['action']['args'])
199                self._current_message = None
200
201                # The return value if any, from an action method is sent back to
202                # the sender as a "response" action.
203                if return_value is not None:
204                    self.send({
205                        "to": message['from'],
206                        "action": {
207                            "name": "response",
208                            "args": {
209                                "data": return_value,
210                                "original_message_id": message.get('id'),
211                            },
212                        }
213                    })
214            else:
215                raise PermissionError(
216                  f"\"{self.id()}.{message['action']['name']}\" not permitted")
217        except Exception as e:
218            error = e  # save the error for after_action
219            raise
220        finally:
221            self.after_action(message, return_value, error)
222
223    def __permitted(self, message: dict) -> bool:
224        """
225        Checks whether the action represented by the message is allowed
226        """
227        action_method = self.__action_method(message['action']['name'])
228        policy = action_method.action_properties["access_policy"]
229        if policy == ACCESS_PERMITTED:
230            return True
231        elif policy == ACCESS_DENIED:
232            return False
233        elif policy == ACCESS_REQUESTED:
234            return self.request_permission(message)
235        else:
236            raise Exception(
237              f"Invalid access policy for method: {message['action']}, got '{policy}'")

239    def __action_methods(self) -> dict:
240        instance_methods = inspect.getmembers(self, inspect.ismethod)
241        action_methods = {
242            method_name: method
243            for method_name, method in instance_methods
244            if hasattr(method, "action_properties")
245        }
246        return action_methods

248    def __action_method(self, action_name: str):
249        """
250        Returns the method for the given action name.
251        """
252        action_methods = self.__action_methods()
253        return action_methods[action_name]

275            or method.action_properties["name"] == action_name
