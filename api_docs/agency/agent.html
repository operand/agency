 25def _generate_help(method: callable) -> dict:
 26    """
 27    Generates a help object from a method's docstring and signature
 28
 29    Args:
 30        method: the method
 31
 32    Returns:
 33        A help object of the form:
 34
 35        {
 36            "description": <description>,
 37            "args": {
 38                "arg_name": {
 39                    "type": <type>,
 40                    "description": <description>
 41                },
 42            }
 43            "returns": {
 44                "type": <type>,
 45                "description": <description>
 46            }
 47        }
 48    """
 49    signature = inspect.signature(method)
 50    parsed_docstring = parse(method.__doc__, DocstringStyle.GOOGLE)
 51
 52    help_object = {}
 53
 54    # description
 55    if parsed_docstring.short_description is not None:
 56        description = parsed_docstring.short_description
 57        if parsed_docstring.long_description is not None:
 58            description += " " + parsed_docstring.long_description
 59        help_object["description"] = re.sub(r"\s+", " ", description).strip()
 60
 61    # args
 62    help_object["args"] = {}
 63    docstring_args = {arg.arg_name: arg for arg in parsed_docstring.params}
 64    arg_names = list(signature.parameters.keys())[1:]  # skip 'self' argument
 65    for arg_name in arg_names:
 66        arg_object = {}
 67
 68        # type
 69        sig_annotation = signature.parameters[arg_name].annotation
 70        if sig_annotation is not None and sig_annotation.__name__ != "_empty":
 71            arg_object["type"] = _python_to_json_type_name(
 72                signature.parameters[arg_name].annotation.__name__)
 73        elif arg_name in docstring_args and docstring_args[arg_name].type_name is not None:
 74            arg_object["type"] = _python_to_json_type_name(
 75                docstring_args[arg_name].type_name)
 76
 77        # description
 78        if arg_name in docstring_args and docstring_args[arg_name].description is not None:
 79            arg_object["description"] = docstring_args[arg_name].description.strip()
 80
 81        help_object["args"][arg_name] = arg_object
 82
 83    # returns
 84    if parsed_docstring.returns is not None:
 85        help_object["returns"] = {}
 86
 87        # type
 88        if signature.return_annotation is not None:
 89            help_object["returns"]["type"] = _python_to_json_type_name(
 90                signature.return_annotation.__name__)
 91        elif parsed_docstring.returns.type_name is not None:
 92            help_object["returns"]["type"] = _python_to_json_type_name(
 93                parsed_docstring.returns.type_name)
 94
 95        # description
 96        if parsed_docstring.returns.description is not None:
 97            help_object["returns"]["description"] = parsed_docstring.returns.description.strip()
 98
 99    return help_object

112def action(*args, **kwargs):
113    """
114    Declares instance methods as actions making them accessible to other agents.
115
116    Keyword arguments:
117        name: The name of the action. Defaults to the name of the method.
118        help: The help object. Defaults to a generated object.
119        access_policy: The access policy. Defaults to ACCESS_PERMITTED.
120    """
121    def decorator(method):
122        action_name = kwargs.get("name", method.__name__)
123        if action_name == _RESPONSE_ACTION_NAME:
124            raise ValueError(f"action name '{action_name}' is reserved")
125        method.action_properties = {
126            "name": method.__name__,
127            "help": _generate_help(method),
128            "access_policy": ACCESS_PERMITTED,
129            **kwargs}
130        return method
131
132    if len(args) == 1 and callable(args[0]) and not kwargs:
133        return decorator(args[0])  # The decorator was used without parentheses
134    else:
135        return decorator  # The decorator was used with parentheses

171                 id: str,
172                 outbound_queue: _QueueProtocol,
173                 receive_own_broadcasts: bool = True):
174        """
175        Initializes an Agent.
176
177        This constructor is not meant to be called directly. It is invoked by
178        the Space class when adding an agent.
179
180        Subclasses should call super().__init__() in their constructor.
181
182        Args:
183            id: The id of the agent
184            outbound_queue: The outgoing queue for sending messages
185            receive_own_broadcasts:
186                Whether the agent will receive its own broadcasts. Defaults to
187                True
188        """
189        if len(id) < 1 or len(id) > 255:
190            raise ValueError("id must be between 1 and 255 characters")
191        if re.match(r"^amq\.", id):
192            raise ValueError("id cannot start with \"amq.\"")
193        if id == "*":
194            raise ValueError("id cannot be \"*\"")
195        if outbound_queue is None:
196            raise ValueError("outbound_queue must be provided")

198        self._outbound_queue: _QueueProtocol = outbound_queue
199        self._receive_own_broadcasts: bool = receive_own_broadcasts
200        self._is_processing: bool = False  # set by the Space
201        self._message_log: List[Message] = []
202        self._message_log_lock = threading.Lock()
203        self._pending_responses: Dict[str, Message] = {}
204        self._pending_responses_lock = threading.Lock()
205        self.__thread_local_current_message = threading.local()
206        self.__thread_local_current_message.value: Message = None

224    def request(self, message: dict, timeout: float = 3) -> object:
225        """
226        Synchronously sends a message then waits for and returns the return
227        value of the invoked action.
228
229        This method allows you to call an action synchronously like a function
230        and receive its return value in python. If the action raises an
231        exception an ActionError will be raised containing the error message.
232
233        Args:
234            message: The message to send
235            timeout:
236                The timeout in seconds to wait for the returned value.
237                Defaults to 3 seconds.
238
239        Returns:
240            object: The return value of the action.
241
242        Raises:
243            TimeoutError: If the timeout is reached
244            ActionError: If the action raised an exception
245        """
246        if not self._is_processing:
247            raise RuntimeError(
248                "request() called while agent is not processing incoming messages. Use send() instead.")
249
250        # Set the message id
251        # Having a meta.request_id identifies it as a request
252        request_id = f"request--{uuid.uuid4()}"
253        message["meta"] = message.get("meta", {})
254        message["meta"]["request_id"] = request_id
255
256        # Send and mark the request as pending
257        self.send(message)
258        pending = object()
259        with self._pending_responses_lock:
260            self._pending_responses[request_id] = pending
261
262        # Wait for response
263        start_time = time.time()
264        while self._pending_responses[request_id] == pending:
265            time.sleep(0.001)
266            if time.time() - start_time > timeout:
267                raise TimeoutError
268
269        # Raise error or return value from response
270        with self._pending_responses_lock:
271            response_message = self._pending_responses.pop(request_id)
272        if "error" in response_message["action"]["args"]:
273            raise ActionError(response_message["action"]["args"]["error"])
274        return response_message["action"]["args"]["value"]

276    def _receive(self, message: dict):
277        """
278        Receives and handles an incoming message.
279
280        Args:
281            message: The incoming message
282        """
283        # Ignore own broadcasts if _receive_own_broadcasts is false
284        if not self._receive_own_broadcasts \
285           and message['from'] == self.id() \
286           and message['to'] == '*':
287            return
288
289        log("debug", f"{self.id()} received message", message)
290
291        # Record the received message before handling
292        with self._message_log_lock:
293            self._message_log.append(message)
294
295        # Handle incoming responses
296        response_id = message.get("meta", {}).get("response_id")
297        if message["action"]["name"] == _RESPONSE_ACTION_NAME:
298            if response_id in self._pending_responses.keys():
299                # This was a response to a request()
300                self._pending_responses[response_id] = message
301                # From here the request() method will pick up the response in
302                # the existing thread
303            else:
304                # This was a response to a send()
305                if "value" in message["action"]["args"]:
306                    handler_callback = self.handle_action_value
307                    arg = message["action"]["args"]["value"]
308                elif "error" in message["action"]["args"]:
309                    handler_callback = self.handle_action_error
310                    arg = ActionError(message["action"]["args"]["error"])
311                else:
312                    raise RuntimeError("Unknown action response")
313
314                # Spawn a thread to handle the response
315                def __process_response(arg, current_message):
316                    log("debug", f"{self.id()} processing response", message)
317                    self.__thread_local_current_message.value = current_message
318                    handler_callback(arg)
319
320                threading.Thread(
321                    target=__process_response,
322                    args=(arg, message, ),
323                    daemon=True,
324                ).start()
325
326        # Handle all other messages
327        else:
328            # Spawn a thread to process the message. This means that messages
329            # are processed concurrently, but may be processed out of order.
330            threading.Thread(
331                target=self.__process, args=(message,), daemon=True).start()
332
333    def __process(self, message: dict):
334        """
335        Top level method within the action processing thread.
336        """
337        log("debug", f"{self.id()} processing message", message)
338        self.__thread_local_current_message.value = message
339        request_id = message.get("meta", {}).get("request_id")
340        response_id = request_id or message.get("meta", {}).get("id")
341        try:
342            # Commit the action
343            return_value = self.__commit(message)
344
345            # If the action returned a value or this was a request (which
346            # expects a value), return it
347            if request_id or return_value is not None:
348                self.send({
349                    "meta": {
350                        "response_id": response_id
351                    },
352                    "to": message['from'],
353                    "action": {
354                        "name": _RESPONSE_ACTION_NAME,
355                        "args": {
356                            "value": return_value,
357                        }
358                    }
359                })
360        except Exception as e:
361            # Handle errors (including PermissionError) that occur while
362            # committing an action by reporting back to the sender.
363            log("debug",
364                f"{self.id()} exception while committing action {message['action']['name']}", e)
365            request_id = message.get("meta", {}).get("request_id")
366            self.send({
367                "meta": {
368                    "response_id": response_id
369                },
370                "to": message['from'],
371                "from": self.id(),
372                "action": {
373                    "name": _RESPONSE_ACTION_NAME,
374                    "args": {
375                        "error": f"{e.__class__.__name__}: {e}"
376                    }
377                }
378            })

380    def __commit(self, message: dict):
381        """
382        Invokes the action method
383
384        Args:
385            message: The incoming message specifying the action
386
387        Raises:
388            AttributeError: If the action method is not found
389            PermissionError: If the action is not permitted
390        """
391        try:
392            # Check if the action method exists
393            action_method = self.__action_method(message["action"]["name"])
394        except KeyError:
395            # the action was not found
396            if message['to'] == '*':
397                return  # broadcasts will not raise an error in this situation
398            else:
399                raise AttributeError(
400                    f"\"{message['action']['name']}\" not found on \"{self.id()}\"")
401
402        # Check if the action is permitted
403        if not self.__permitted(message):
404            raise PermissionError(
405                f"\"{self.id()}.{message['action']['name']}\" not permitted")
406
407        self.before_action(message)
408
409        return_value = None
410        error = None
411        try:
412            # Invoke the action method
413            return_value = action_method(**message['action'].get('args', {}))
414        except Exception as e:
415            error = e
416            raise
417        finally:
418            self.after_action(message, return_value, error)
419        return return_value

421    def __permitted(self, message: dict) -> bool:
422        """
423        Checks whether the message's action is allowed
424        """
425        action_method = self.__action_method(message['action']['name'])
426        policy = action_method.action_properties["access_policy"]
427        if policy == ACCESS_PERMITTED:
428            return True
429        elif policy == ACCESS_DENIED:
430            return False
431        elif policy == ACCESS_REQUESTED:
432            return self.request_permission(message)
433        else:
434            raise Exception(
435              f"Invalid access policy for method: {message['action']}, got '{policy}'")

252        # Having a meta.request_id identifies it as a request
253        request_id = f"request--{uuid.uuid4()}"
254        message["meta"] = message.get("meta", {})
255        message["meta"]["request_id"] = request_id
256
257        # Send and mark the request as pending
258        self.send(message)
259        pending = object()
260        with self._pending_responses_lock:
261            self._pending_responses[request_id] = pending
262
263        # Wait for response
264        start_time = time.time()
265        while self._pending_responses[request_id] == pending:
266            time.sleep(0.001)
267            if time.time() - start_time > timeout:
268                raise TimeoutError
269
270        # Raise error or return value from response
271        with self._pending_responses_lock:
272            response_message = self._pending_responses.pop(request_id)
273        if "error" in response_message["action"]["args"]:
274            raise ActionError(response_message["action"]["args"]["error"])
275        return response_message["action"]["args"]["value"]

277    def _receive(self, message: dict):
278        """
279        Receives and handles an incoming message.
280
281        Args:
282            message: The incoming message
283        """
284        # Ignore own broadcasts if _receive_own_broadcasts is false
285        if not self._receive_own_broadcasts \
286           and message['from'] == self.id() \
287           and message['to'] == '*':
288            return
289
290        log("debug", f"{self.id()} received message", message)
291
292        # Record the received message before handling
293        with self._message_log_lock:
294            self._message_log.append(message)
295
296        # Handle incoming responses
297        response_id = message.get("meta", {}).get("response_id")
298        if message["action"]["name"] == _RESPONSE_ACTION_NAME:
299            if response_id in self._pending_responses.keys():
300                # This was a response to a request()
301                self._pending_responses[response_id] = message
302                # From here the request() method will pick up the response in
303                # the existing thread
304            else:
305                # This was a response to a send()
306                if "value" in message["action"]["args"]:
307                    handler_callback = self.handle_action_value
308                    arg = message["action"]["args"]["value"]
309                elif "error" in message["action"]["args"]:
310                    handler_callback = self.handle_action_error
311                    arg = ActionError(message["action"]["args"]["error"])
312                else:
313                    raise RuntimeError("Unknown action response")
314
315                # Spawn a thread to handle the response
316                def __process_response(arg, current_message):
317                    log("debug", f"{self.id()} processing response", message)
318                    self.__thread_local_current_message.value = current_message
319                    handler_callback(arg)
320
321                threading.Thread(
322                    target=__process_response,
323                    args=(arg, message, ),
324                    daemon=True,
325                ).start()
326
327        # Handle all other messages
328        else:
329            # Spawn a thread to process the message. This means that messages
330            # are processed concurrently, but may be processed out of order.
331            threading.Thread(
332                target=self.__process, args=(message,), daemon=True).start()

334    def __process(self, message: dict):
335        """
336        Top level method within the action processing thread.
337        """
338        log("debug", f"{self.id()} processing message", message)
339        self.__thread_local_current_message.value = message
340        request_id = message.get("meta", {}).get("request_id")
341        response_id = request_id or message.get("meta", {}).get("id")
342        try:
343            # Commit the action
344            return_value = self.__commit(message)
345
346            # If the action returned a value or this was a request (which
347            # expects a value), return it
348            if request_id or return_value is not None:
349                self.send({
350                    "meta": {
351                        "response_id": response_id
352                    },
353                    "to": message['from'],
354                    "action": {
355                        "name": _RESPONSE_ACTION_NAME,
356                        "args": {
357                            "value": return_value,
358                        }
359                    }
360                })
361        except Exception as e:
362            # Handle errors (including PermissionError) that occur while
363            # committing an action by reporting back to the sender.
364            log("debug",
365                f"{self.id()} exception while committing action {message['action']['name']}", e)
366            request_id = message.get("meta", {}).get("request_id")
367            self.send({
368                "meta": {
369                    "response_id": response_id
370                },
371                "to": message['from'],
372                "from": self.id(),
373                "action": {
374                    "name": _RESPONSE_ACTION_NAME,
375                    "args": {
376                        "error": f"{e.__class__.__name__}: {e}"
377                    }
378                }
379            })

381    def __commit(self, message: dict):
382        """
383        Invokes the action method
384
385        Args:
386            message: The incoming message specifying the action
387
388        Raises:
389            AttributeError: If the action method is not found
390            PermissionError: If the action is not permitted
391        """
392        try:
393            # Check if the action method exists
394            action_method = self.__action_method(message["action"]["name"])
395        except KeyError:
396            # the action was not found
397            if message['to'] == '*':
398                return  # broadcasts will not raise an error in this situation
399            else:
400                raise AttributeError(
401                    f"\"{message['action']['name']}\" not found on \"{self.id()}\"")
402
403        # Check if the action is permitted
404        if not self.__permitted(message):
405            raise PermissionError(
406                f"\"{self.id()}.{message['action']['name']}\" not permitted")
407
408        self.before_action(message)
409
410        return_value = None
411        error = None
412        try:
413            # Invoke the action method
414            return_value = action_method(**message['action'].get('args', {}))
415        except Exception as e:
416            error = e
417            raise
418        finally:
419            self.after_action(message, return_value, error)
420        return return_value

422    def __permitted(self, message: dict) -> bool:
423        """
424        Checks whether the message's action is allowed
425        """
426        action_method = self.__action_method(message['action']['name'])
427        policy = action_method.action_properties["access_policy"]
428        if policy == ACCESS_PERMITTED:
429            return True
430        elif policy == ACCESS_DENIED:
431            return False
432        elif policy == ACCESS_REQUESTED:
433            return self.request_permission(message)
434        else:
435            raise Exception(
436              f"Invalid access policy for method: {message['action']}, got '{policy}'")
